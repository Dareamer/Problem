主要用在页面字段验证上。  

项目中用到的正则：  
````  
"*":"不能为空！",
			"*6-16":"请填写6到16位任意字符！",
			"n":"请填写数字！",
			"n6-16":"请填写6到16位数字！",
			"s":"不能输入特殊字符！",
			"s6-18":"请填写6到18位字符！",
			"p":"请填写邮政编码！",
			"m":"请填写手机号码！",
			"e":"邮箱地址格式不对！",
			"url":"请填写网址！",
			"earth":"请输入正确的数字！(保留2位小数，最多9位)", //不能为空
			"num2":"请输入正确的数字！(保留2位小数，最多9位)",   //可以为空
			"ph":"请输入手机号码或固话!",
			"negativeNum":"请输入正确的数字！(保留2位小数，最多9位)" //不能为空, 正数 或 负数  
      
      "*":/[\w\W]+/,
			"*6-16":/^[\w\W]{6,16}$/,
			"n":/^\d+$/,
			"n6-16":/^\d{6,16}$/,
			"s":/^[\u4E00-\u9FA5\uf900-\ufa2d\w\.\s]+$/,
			"s6-18":/^[\u4E00-\u9FA5\uf900-\ufa2d\w\.\s]{6,18}$/,
			"p":/^[0-9]{6}$/,
			"m":/^13[0-9]{9}$|14[0-9]{9}|15[0-9]{9}$|18[0-9]{9}$/,
			"e":/^\w+([-+.']\w+)*@\w+([-.]\w+)*\.\w+([-.]\w+)*$/,
			"url":/^(\w+:\/\/)?\w+(\.\w+)+.*$/,
			"earth":/(^[1-9][0-9]{0,8}(\.[0-9]{1,2})?$)|(^(0){1}$)|(^[0-9]\.[0-9]([0-9])?$)/,
			"num2":/^\s*$|(^[1-9][0-9]{0,8}(\.[0-9]{1,2})?$)|(^(0){1}$)|(^[0-9]\.[0-9]([0-9])?$)/,
			///(^([0-9]{3,4}-)?[0-9]{7,8}$)|(^((\(\d{3}\))|(\d{3}\-))?(1[34578]\d{9})$)|^\s*$/
			"ph": /^\s*$|^[\d-]{0,100}$/,
			"negativeNum":/(^\-?[1-9][0-9]{0,8}(\.[0-9]{1,2})?$)|(^(0){1}$)|(^\-?[0-9]\.[0-9]([0-9])?$)/
````
